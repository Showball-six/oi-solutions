P2322 [HNOI2006] 最短母串问题

状压DP+KMP

思路：

求出一个最短字符串要包含 n 个给出的字符串。如果不考虑长度，把 n 个给出的字符串拼接起来即可。为了使长度更短。容易发现如 s="ABCD", t=BCDABC。那么拼接在一起，只需要 ABCDABC 即可。s 的后缀和 t 的前缀的最长公共部分可以省去。

不妨令 gi,j 表示 i 串和 j 串拼接的最长公共部分。可以通过 KMP 预处理。

考虑状压 DP。令 fi,j 表示当前状态为 i，最后拼接的字符串为 j 时，最优的字符串。（因为题目要求字典序最小，DP数组直接存字符串）

转移时，枚举所有状态和前一个状态的结尾字符串，然后去掉重合部分拼接，并更新 DP 值即可。

最后，扫描一遍以每个字符串结尾的 DP 值，更新答案即可。

坑点：

  • 如果给出的一个字符串是其它字符串的子串，那么就不需要考虑，因为其它字符串被包含，它一定也被包含。
  • 考虑以上情况后，必须先对给出的字符串去重，因为如果两个字符串相等，按照以上情况处理后，两个字符串均不会被考虑进去。

具体实现参考代码及注释

代码：

#include<bits/stdc++.h>


using namespace std;


using i64=long long;


const int N=13;


//KMP模板
template<class Type>
struct KMP{
    vector<int> init(Type s){
        int n=(int)s.size()-1;
        vector<int> nxt(n+1,0);
        for(int i=2;i<=n;i++){
            nxt[i]=nxt[i-1];
            while(nxt[i]&&s[nxt[i]+1]!=s[i]) nxt[i]=nxt[nxt[i]];
            nxt[i]+=(s[nxt[i]+1]==s[i]);
        }
        return nxt;
    }


    vector<int> match(Type s,Type t){
        vector<int> pos;
        vector<int> nxt=init(t);
        int n=(int)s.size()-1,m=(int)t.size()-1;
        for(int i=1,j=0;i<=n;i++){
            while(j&&s[i]!=t[j+1]) j=nxt[j];
            j+=(t[j+1]==s[i]);
            if(j==m){//匹配成功
                j=nxt[j];
                pos.push_back(i-m+1);
            }
        }
        return pos;
    }
};
KMP<string> kmp;


//DP数组
string f[1<<N][N];
void Showball(){
    int n;
    cin>>n;
    vector<string> s(n);
    for(int i=0;i<n;i++) cin>>s[i];


    //排序去重字符串
    sort(s.begin(),s.end());
    s.erase(unique(s.begin(),s.end()),s.end());
    n=s.size();


    //筛选出不是其它字符串子串的字符串
    vector<string> cand;
    for(int i=0;i<n;i++){
        bool ok=true;
        for(int j=0;j<n;j++){
            if(i!=j&&s[j].find(s[i])!=s[j].npos){
                ok=false;
                break;
            }
        }
        if(ok) cand.push_back("?"+s[i]);
    }
    n=cand.size();


    //求出两个字符串拼接后的最长公共部分
    auto get=[&](string s,string t){
        string tmp=t+s;
        return kmp.init(tmp).back();
    };


    //维护g[i][j]
    vector<vector<int>> g(n,vector<int>(n));
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(i!=j) g[i][j]=get(cand[i],cand[j]);
        }
    }


    //初始化dp数组
    for(int i=0;i<n;i++){
        f[1<<i][i]=cand[i].substr(1);
    }




    for(int i=0;i<(1<<n);i++){//枚举所有状态
        for(int j=0;j<n;j++){//枚举最后一个拼接的字符串
            if(i>>j&1){//状态中包含j
                for(int k=0;k<n;k++){//枚举前一个状态结尾字符串
                    if(j!=k&&i>>k&1){
                        string t=cand[j].substr(g[k][j]+1);//k和j拼接去除最长公共部分后的字符串
                        t=f[i-(1<<j)][k]+t;//拼接
                        //转移，更新dp值
                        if(f[i][j].size()==0){
                            f[i][j]=t;
                        }else{
                            if(f[i][j].size()>t.size()) f[i][j]=t;
                            else if(f[i][j].size()==t.size()&&f[i][j]>t) f[i][j]=t;
                        }
                    }
                }
            }
        }
    }


    //维护答案
    int tt=(1<<n)-1;
    string ans="";
    for(int i=0;i<n;i++){
        if(ans.size()==0) ans=f[tt][i];
        else{
            if(ans.size()>f[tt][i].size()) ans=f[tt][i];
            else if(ans.size()==f[tt][i].size()&&f[tt][i]<ans) ans=f[tt][i];
        }
    }
    cout<<ans<<"\n";
}
int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t=1;
    //cin>>t;


    while(t--){
      Showball();
    }


    return 0;
}

 

ACAM+BFS

思路：

建立 ACAM，给每个表示字符串的节点进行状态压缩标记（第 i 个字符串标记为 2i）。

然后沿着 Fail 合并标记。那么，就可以在 Trie 上进行 BFS。

如果当前字符串状态为：2n−1。说明包含了所有的字符串，并且是字典序最小的解。

代码：

xxxxxxxxxx
#include<bits/stdc++.h>


using namespace std;


using i64=long long;


const int N=610,M=(1<<12)+10;
int ch[N][26],cnt[N],idx;
int ne[N],ed[N];


//ACAM
void insert(string s,int v){//建树 
    int p=0;
    for(auto c:s){
        int i=c-'A';
        if(!ch[p][i]) ch[p][i]=++idx;
        p=ch[p][i];
    } 
    ed[p]|=1<<v;
}


void build(){//建ACAM 
    queue<int> q;
    for(int i=0;i<26;i++) if(ch[0][i]) q.push(ch[0][i]);
    while(q.size()){
        int u=q.front();q.pop();
        for(int i=0;i<26;i++){
            int v=ch[u][i];
            if(v) ne[v]=ch[ne[u]][i],ed[v]|=ed[ne[v]],q.push(v);
            else ch[u][i]=ch[ne[u]][i];
        }
    } 
}


int vis[N][M],pre[N*M];
char ans[N*M];


void Showball(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
        string s;
        cin>>s;
        insert(s,i);
    }
    
    build();
    
    int all=(1<<n)-1;


    queue<pair<int,int>> q;
    q.push({0,0});
    vis[0][0]=1;
    int cnt=0,cur=0;
    while(q.size()){
        auto [u,st]=q.front();
        q.pop();
        if(st==all){
            string res="";
            while(cur){
                res+=ans[cur];
                cur=pre[cur];
            }
            reverse(res.begin(),res.end());
            return cout<<res<<"\n",void();
        }


        for(int i=0;i<26;i++){
            int v=ch[u][i];
            if(!vis[v][st|ed[v]]){
                vis[v][st|ed[v]]=1;
                q.push({v,st|ed[v]});
                pre[++cnt]=cur;
                ans[cnt]=char(i+'A');
            }
        }
        cur++;
    }
}
int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t=1;
    //cin>>t;
    while(t--){
        Showball();
    }
    return 0;
}
